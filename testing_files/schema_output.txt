--- Schema for customers ---
Column: customer_id, Type: VARCHAR(50)
Column: name, Type: VARCHAR(100)
Column: email, Type: VARCHAR(100)
Column: phone_number, Type: VARCHAR(15)
Column: address, Type: VARCHAR(200)
Column: service_plan_id, Type: VARCHAR(50)
Column: account_status, Type: VARCHAR(20)
Column: registration_date, Type: DATE
Column: last_billing_date, Type: DATE
